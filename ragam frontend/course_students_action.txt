// GET COURSE STUDENTS - Returns list of enrolled students for a course  
if ($action == 'course_students') {
    if (!isset($_GET['course_id'])) {
        echo json_encode(['success' => false, 'message' => 'Course ID is required']);
        exit;
    }
    $courseId = intval($_GET['course_id']);
    // Get database connection from config
    require_once 'config.php';
    $database = new Database();
    $conn = $database->getConnection();
    try {
        // Get all students enrolled in this course
        $query = "SELECT u.user_id, u.username, u.email, e.enrolled_at,
                         (SELECT COUNT(*) FROM quiz_results qr WHERE qr.student_id = u.user_id AND qr.course_id = e.course_id) as quiz_attempts,
                         (SELECT MAX(qr.score) FROM quiz_results qr WHERE qr.student_id = u.user_id AND qr.course_id = e.course_id) as best_score
                  FROM enrollments e
                  INNER JOIN users u ON e.student_id = u.user_id
                  WHERE e.course_id = :course_id
                  ORDER BY e.enrolled_at DESC";
        $stmt = $conn->prepare($query);
        $stmt->bindParam(':course_id', $courseId, PDO::PARAM_INT);
        $stmt->execute();
        $students = $stmt->fetchAll(PDO::FETCH_ASSOC);
        echo json_encode([
            'success' => true,
            'data' => $students,
            'count' => count($students)
        ]);
    } catch (PDOException $e) {
        echo json_encode(['success' => false, 'message' => 'Database error: ' . $e->getMessage()]);
    }
    exit;
}
